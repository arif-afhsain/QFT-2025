    [p, p]~~ &\sim ~~0 \\
    [J, J]~~ &\sim ~J \\
    [J, p^2]~ &\sim ~0 \\
    [J, p_ap^a] &= [J, p_a]p^a + p_a[J, p^a]\sim 0
\end{align*}
